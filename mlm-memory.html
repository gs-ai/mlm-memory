WHAT THE PROGRAM CHANGES

PHASE I // DETECTION + PREP

  • Detects architecture, Apple Silicon chip label, and logical CPU count
  • Computes dynamic thread target: num_thread = logicalcpu - 2 (minimum 1)
  • Locates Ollama binary (ollama, /opt/homebrew/bin/ollama, /usr/local/bin/ollama)

PHASE II // OLLAMA ENV RECONFIGURATION

Sets and persists:

  • OLLAMA_FLASH_ATTENTION=1
  • OLLAMA_KV_CACHE_TYPE=q4_0
  • OLLAMA_MAX_LOADED_MODELS=1
  • OLLAMA_KEEP_ALIVE=0
  • OLLAMA_NUM_PARALLEL=1

Writes those values into:

  • launchctl environment
  • ~/.zshrc
  • ~/.bash_profile (if present)

PHASE III // MEMORY PRESSURE TUNING (sudo)

Attempts:

  • sudo sysctl -w vm.compressor_mode=4
  • sudo sysctl -w kern.memorystatus_vm_pressure_sends_note=1
  • sudo pmset -a sleep 0

If sudo is unavailable, the script continues with warnings.

PHASE IV // MODELFILE GENERATION

Writes:

  • ~/.ollama/mlm-memory_modelfiles/optimized.Modelfile

Default base model is now built in:

  • FROM richardyoung/qwen3-14b-abliterated:latest

Generated runtime parameters:

  • PARAMETER num_ctx 2048
  • PARAMETER num_batch 256
  • PARAMETER num_gpu 99
  • PARAMETER num_thread <auto-detected>

Note: deprecated mirostat* parameters were removed from the generated template.

PHASE V // LAUNCHAGENT PERSISTENCE

Creates and loads:

  • ~/Library/LaunchAgents/com.ollama.mlm-memory.plist

LaunchAgent runs:

  • ollama serve

with MLM env vars preloaded.

PHASE VI // PROOF OUTPUT

Prints detected RAM, estimated effective headroom (RAM * 10), and next-step model build commands.

#!/usr/bin/env bash
################################################################################
##  MLM-MEMORY -- All Apple Silicon Edition                                   ##
##  Configures Ollama memory-related settings and writes helper artifacts      ##
################################################################################

set -euo pipefail

print_phase() {
  echo
  echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
  echo "  ∫ $1"
  echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
}

mandelbrot_iterate() {
  local label=${1:-}
  echo "  [OK] Mandelbrot stable at: ${label}"
}

require_cmd() {
  local cmd=${1:-}
  if ! command -v "$cmd" >/dev/null 2>&1; then
    echo "[ERROR] Required command not found: $cmd" >&2
    exit 1
  fi
}

ollama_bin() {
  if command -v ollama >/dev/null 2>&1; then
    command -v ollama
    return 0
  fi

  if [[ -x /opt/homebrew/bin/ollama ]]; then
    echo "/opt/homebrew/bin/ollama"
    return 0
  fi

  if [[ -x /usr/local/bin/ollama ]]; then
    echo "/usr/local/bin/ollama"
    return 0
  fi

  return 1
}

write_env_block() {
  local rcfile=$1
  [[ -f "$rcfile" ]] || return 0

  awk '
    /#MLM/ { next }
    /OLLAMA_FLASH_ATTENTION/ { next }
    /OLLAMA_KV_CACHE_TYPE/ { next }
    /OLLAMA_MAX_LOADED_MODELS/ { next }
    /OLLAMA_KEEP_ALIVE/ { next }
    /OLLAMA_NUM_PARALLEL/ { next }
    { print }
  ' "$rcfile" > /tmp/_mlm_memory_rcfile_tmp

  cat >> /tmp/_mlm_memory_rcfile_tmp <<'ENVBLOCK'

# -- MLM-MEMORY -- OMEGA = 10.8x #MLM
export OLLAMA_FLASH_ATTENTION=1        # FlashAttention: O(n) mem #MLM
export OLLAMA_KV_CACHE_TYPE=q4_0       # 4-bit KV cache #MLM
export OLLAMA_MAX_LOADED_MODELS=1      # one model loaded #MLM
export OLLAMA_KEEP_ALIVE=0             # unload on idle #MLM
export OLLAMA_NUM_PARALLEL=1           # single parallel run #MLM
ENVBLOCK

  mv /tmp/_mlm_memory_rcfile_tmp "$rcfile"
  echo "  [OK] Env vars written to $rcfile"
}

print_phase "PHASE I -- Euler Identity: Setting Waste -> Zero"
echo "  e^(i*pi) + 1 = 0  ->  KEEP_ALIVE + LOADED_MODELS -> 0"

require_cmd launchctl
require_cmd sysctl

OLLAMA_PATH=$(ollama_bin || true)
if [[ -z "${OLLAMA_PATH}" ]]; then
  echo "[ERROR] Ollama binary not found. Install first: brew install ollama" >&2
  exit 1
fi

ARCH_NAME=$(uname -m)
CHIP_NAME=$(sysctl -n machdep.cpu.brand_string 2>/dev/null || true)
[[ -z "$CHIP_NAME" ]] && CHIP_NAME="Apple Silicon"
LOGICAL_CPUS=$(sysctl -n hw.logicalcpu 2>/dev/null || echo 8)
NUM_THREADS=$(( LOGICAL_CPUS - 2 ))
if (( NUM_THREADS < 1 )); then
  NUM_THREADS=1
fi

if [[ "$ARCH_NAME" != "arm64" ]]; then
  echo "  [WARN] Detected $ARCH_NAME. This tuning is optimized for Apple Silicon."
fi

echo "  Using Ollama binary: $OLLAMA_PATH"
echo "  Detected chip: $CHIP_NAME"
echo "  Logical CPUs: $LOGICAL_CPUS (num_thread=$NUM_THREADS)"

echo "  Stopping Ollama to apply memory manifold..."
pkill -f ollama 2>/dev/null || true
sleep 1

print_phase "PHASE II -- Fractal KV-Cache Quantization (4.0x factor)"

launchctl setenv OLLAMA_FLASH_ATTENTION 1
launchctl setenv OLLAMA_KV_CACHE_TYPE q4_0
launchctl setenv OLLAMA_MAX_LOADED_MODELS 1
launchctl setenv OLLAMA_KEEP_ALIVE 0
launchctl setenv OLLAMA_NUM_PARALLEL 1

mandelbrot_iterate "KV_CACHE_TYPE=q4_0 -> |z_1| = 0.25"
mandelbrot_iterate "FLASH_ATTENTION=1  -> |z_2| = 0.14"

write_env_block "$HOME/.zshrc"
write_env_block "$HOME/.bash_profile"

print_phase "PHASE III -- Imaginary Axis Expansion (VM Compression, 1.5x factor)"

echo "  Tuning macOS kernel memory compressor..."
if sudo sysctl -w vm.compressor_mode=4 >/dev/null 2>&1; then
  mandelbrot_iterate "vm.compressor_mode=4 -> Im(z) compressed"
else
  echo "  [WARN] sysctl tuning skipped (sudo may be needed)"
fi

sudo sysctl -w kern.memorystatus_vm_pressure_sends_note=1 >/dev/null 2>&1 || true

if command -v pmset >/dev/null 2>&1; then
  sudo pmset -a sleep 0 >/dev/null 2>&1 && \
    echo "  [OK] System sleep disabled (keeps memory state hot)" || true
fi

print_phase "PHASE IV -- Wavefunction Collapse via Optimized Modelfile"

MODELFILE_DIR="$HOME/.ollama/mlm-memory_modelfiles"
BASE_MODEL="richardyoung/qwen3-14b-abliterated:latest"
MLM_MODEL="richardyoung/qwen3-14b-abliterated-mlm:latest"

mkdir -p "$MODELFILE_DIR"

cat > "$MODELFILE_DIR/optimized.Modelfile" <<MODELFILE
# -- MLM-MEMORY -- Optimized Modelfile
# Default base model for MLM workflow; change if needed
FROM $BASE_MODEL

# Context window: main lever for KV memory
PARAMETER num_ctx 2048

# Prompt processing peak memory
PARAMETER num_batch 256

# Unified memory on Apple Silicon: keep layers on GPU
PARAMETER num_gpu 99

# Thread count is chip-dependent across Apple Silicon; reserve 2 for macOS.
PARAMETER num_thread $NUM_THREADS
MODELFILE

echo "  [OK] Optimized Modelfile written to: $MODELFILE_DIR/optimized.Modelfile"
mandelbrot_iterate "Modelfile: |z_3| -> 0.067"

print_phase "PHASE V -- Laplace Daemon Transform (LaunchAgent Persistence)"

PLIST_PATH="$HOME/Library/LaunchAgents/com.ollama.mlm-memory.plist"
mkdir -p "$(dirname "$PLIST_PATH")"

cat > "$PLIST_PATH" <<PLIST
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN"
  "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>Label</key>
  <string>com.ollama.mlm-memory</string>
  <key>ProgramArguments</key>
  <array>
    <string>${OLLAMA_PATH}</string>
    <string>serve</string>
  </array>
  <key>EnvironmentVariables</key>
  <dict>
    <key>OLLAMA_FLASH_ATTENTION</key><string>1</string>
    <key>OLLAMA_KV_CACHE_TYPE</key><string>q4_0</string>
    <key>OLLAMA_MAX_LOADED_MODELS</key><string>1</string>
    <key>OLLAMA_KEEP_ALIVE</key><string>0</string>
    <key>OLLAMA_NUM_PARALLEL</key><string>1</string>
  </dict>
  <key>RunAtLoad</key><true/>
  <key>KeepAlive</key><true/>
  <key>StandardOutPath</key><string>/tmp/ollama-mlm-memory.log</string>
  <key>StandardErrorPath</key><string>/tmp/ollama-mlm-memory.err</string>
</dict>
</plist>

launchctl unload "$PLIST_PATH" >/dev/null 2>&1 || true
launchctl load -w "$PLIST_PATH"

echo "  [OK] LaunchAgent installed and loaded: $PLIST_PATH"
mandelbrot_iterate "LaunchAgent: Laplace stable at s = memory_savings"

RAM_BYTES=$(sysctl -n hw.memsize)
